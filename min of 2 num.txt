set serveroutput on
declare 
a number;
b number;
c number;
 PROCEDURE findMin(x in number,y in number,z out number) is 
begin
if x<y then
z:=x;
else
z:=y;
end if;
end;

begin
a:=&a;
b:=&b;
findMin(a,b,c);
dbms_output.put_line('MINIMUM'||c);
end; 
/
