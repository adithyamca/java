set serveroutput on
declare 
a number;
b number;
c number;
 FUNCTION findMax(x in number,y in number) 
RETURN number
is
z number;

begin
if x>y then
z:=x;
else
z:=y;
end if;
return z;
end;

begin
a:=&a;
b:=&b;
c:=findMax(a,b);
dbms_output.put_line('Maximum'||c);
end; 
/
